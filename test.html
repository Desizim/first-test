    function checkAnswers() {

        var correctAnswers = [1, 2, 1]; // правильные ответы (1-й вопрос - 1-й ответ, 2-й вопрос - 2-й ответ и т.д.)

        var userAnswers = [];

        var resultText = '';


        // Сбор ответов пользователя

        var q1 = document.querySelector('input[name="q1"]:checked');

        var q2 = document.querySelector('input[name="q2"]:checked');

        var q3 = document.querySelector('input[name="q3"]:checked');

        

        if (q1) userAnswers.push(parseInt(q1.value));

        if (q2) userAnswers.push(parseInt(q2.value));

        if (q3) userAnswers.push(parseInt(q3.value));


        // Проверка ответов

        var correctCount = 0;

        for (var i = 0; i < correctAnswers.length; i++) {

            if (userAnswers[i] === correctAnswers[i]) {

                correctCount++;

            }

        }


        // Вывод результатов

        if (correctCount === correctAnswers.length) {

            resultText = "Поздравляем! Вы прошли тест!";

        } else {

            resultText = "Вы ответили правильно на " + correctCount + " из 3 вопросов.";

        }


        document.getElementById("result").innerText = resultText;

    }

</script>
